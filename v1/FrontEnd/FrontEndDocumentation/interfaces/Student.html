export interface Student {
    student_id?: number,           // ID of the student.
    student_name?: string,        // Full name of the student.
    student_document?: string,    // Document (e.g. ID card, passport) number of the student.
    student_age?: string          // Age of the student.
    student_birthdate?: string,   // Date of birth of the student.
    student_country?: string,     // Country of origin of the student.
    student_state?: string,       // State of origin of the student.
    student_city?: string,        // City of origin of the student.
    student_adress?: string,      // Address of the student.
    student_zipcode?: string,     // Postal code of the student's address.
    student_email?: string,       // Email address of the student.
    student_phone?: string,       // Phone number of the student.
    student_phone_2?: string,     // Alternative phone number of the student.
    student_contract_term?: boolean,  // Flag indicating whether the student has agreed to the terms of the contract.
    student_creation_date?: string    // Date when the student was added to the system.
}
