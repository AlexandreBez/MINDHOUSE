export interface Teacher {
  teacher_id?: number; // ID of the teacher.
  teacher_name?: string; // Full name of the teacher.
  teacher_document?: string; // Document (e.g. ID card, passport) number of the teacher.
  teacher_age?: string; // Age of the teacher.
  teacher_birthdate?: string; // Date of birth of the teacher.
  teacher_country?: string; // Country of origin of the teacher.
  teacher_state?: string; // State of origin of the teacher.
  teacher_city?: string; // City of origin of the teacher.
  teacher_address?: string; // Address of the teacher.
  teacher_zipcode?: string; // Postal code of the teacher's address.
  teacher_email?: string; // Email address of the teacher.
  teacher_phone?: string; // Phone number of the teacher.
  teacher_phone_2?: string; // Alternative phone number of the teacher.
  teacher_creation_date?: string; // Date when the teacher was added to the system.
}
