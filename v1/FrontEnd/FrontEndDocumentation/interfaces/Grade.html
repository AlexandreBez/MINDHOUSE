import { Classroom } from './Classroom';
import { Student } from './Student';

/**
 * Represents a grade object.
 * @interface
 */
export interface Grade {
  grade_id?: number; // ID of the grade.
  fk_classroom?: Classroom; // ID of the classroom.
  fk_student?: Student; // ID of the student.
  first_note?: number; // First note of the grade.
  second_note?: number; // Second note of the grade.
  third_note?: number; // Third note of the grade.
  additional_note?: number; // Additional note of the grade.
  final_note?: number; // Final note of the grade.
  status?: string; // Status of the grade.
  grade_creation_date?: string; // Date when the grade was created.
}
