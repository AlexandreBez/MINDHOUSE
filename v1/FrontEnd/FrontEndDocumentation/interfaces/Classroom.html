import { Teacher } from "./Teacher";
import { Course } from "./Course";

/**
 * Represents a classroom object.
 * @interface]
 */
export interface Classroom {
    classroom_id?: number; // ID of the classroom.
    fk_teacher?: Teacher; // ID of the teacher.
    fk_course?: Course; // ID of the course.
    classroom_period?: string; // Period of the classroom (e.g. morning, afternoon, evening).
    classroom_start_time?: string; // Start time of the classroom.
    classroom_end_time?: string; // End time of the classroom.
    classroom_start_date?: string; // Start date of the classroom.
    classroom_end_date?: string; // End date of the classroom.
    classroom_limit?: number; // Limit of students in the classroom.
    classroom_qtd_students?: number; // Quantity of students currently enrolled in the classroom.
    classroom_creation_date?: string; // Date when the classroom was created.
    }
